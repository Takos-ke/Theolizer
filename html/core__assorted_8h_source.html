150 template<class tPolyRangeBase, class tValueType>
151 class PolyReferencer
152 {
153  tPolyRangeBase& mPolyRange;
154 public:
155  PolyReferencer(tPolyRangeBase& iPolyRange) : mPolyRange(iPolyRange)
156  { }
157 
158  tValueType& operator*()
159  {
160  THEOLIZER_INTERNAL_ASSERT(mPolyRange.mHolder.get(), THEOLIZER_INTERNAL_BUG);
161  return mPolyRange.mHolder->front();
162  }
163  tValueType const& operator*() const
164  {
165  THEOLIZER_INTERNAL_ASSERT(mPolyRange.mHolder.get(), THEOLIZER_INTERNAL_BUG);
166  return mPolyRange.mHolder->front();
167  }
168  void operator++()
169  {
170  THEOLIZER_INTERNAL_ASSERT(mPolyRange.mHolder.get(), THEOLIZER_INTERNAL_BUG);
171  mPolyRange.mHolder->drop_front();
172  }
173  bool operator!=(PolyReferencer const& iRhs) const
174  {
175  if (!mPolyRange.mHolder.get())
176  return false;
177  return !mPolyRange.mHolder->empty();
178  }
179 };

206  template<typename tIterator>
207  struct Holder : public HolderBase
208  {
209  tIterator mBegin;
210  tIterator mEnd;
211  std::size_t mCount;
212 
213  Holder(tIterator&& iBegin, tIterator&& iEnd, std::size_t iCount) :
214  mBegin(std::forward<tIterator>(iBegin)),
215  mEnd (std::forward<tIterator>(iEnd)),
216  mCount(iCount)
217  { }
218  ~Holder() = default;
219 
220  tValueType& front()
221  {
222  return *mBegin;
223  }
224  tValueType const& front() const
225  {
226  return *mBegin;
227  }
228  void drop_front()
229  {
230  ++mBegin;
231  --mCount;
232  }
233  bool empty() const
234  {
235  return mBegin == mEnd;
236  }
237  std::size_t size()
238  {
239  return mCount;
240  }
241  };

281 // 生配列の型定義
282 // Array<型, 要素数...>にて生配列の型になる
283 // 下記2つは同じ定義である。
284 // theolizer::Array<int, 10, 20> aIntArray;
285 // int aIntArray[10][20];
286 // このテンプレートで配列の参照を下記のように定義することができる。
287 // theolizer::Array<int, 10>& aInt10;
288 // int (&aInt)[10];
289 // CoveredArray<型, 要素数...>は上記と同じ記憶領域配置だが型が異なる
290 // CoveredArray<int, 2, 3>は下記
291 // class ArrayImpl // == CoveredArray
292 // {
293 // class ArrayImpl
294 // {
295 // class ArrayImpl
296 // {
297 // int mElement;
298 // } mElements[3];
299 // } mElements[2];
300 // };
301 // ***************************************************************************
302 
303 template<class tType, size_t... tDims>
304 struct ArrayImpl
305 {
306  typedef tType type;
307  typedef tType CoveredElements;
308 
309  // オフセットを合わせるため、中身を定義
310  CoveredElements mElement;
311 
312  // 中身へのアクセス
313  CoveredElements& getElement() {return mElement;}
314 };
315 
316 template<class tType, size_t tFirst, size_t... tDims>
317 struct ArrayImpl<tType, tFirst, tDims...>
318 {
319  typedef typename ArrayImpl<tType, tDims...>::type type[tFirst];
320  typedef ArrayImpl<tType, tDims...> CoveredElements[tFirst];
321 
322  // オフセットを合わせるため、中身を定義
323  CoveredElements mElements;
324 
325  // 1次元配列としてアクセス
326  CoveredElements& getElements() {return mElements;}
327 };

596  }
597  }

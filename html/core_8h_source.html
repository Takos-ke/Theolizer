371  {
372  BaseSerializer::AutoRestoreSave aAutoRestoreSave
373  (
374  iSerializer,
375  theolizer::internal::emOrder,
376  true
377  );
378  bool aIsSaved;
379  theolizer::internal::SerializeInfo& aSerializeInfo=
380  iSerializer.registerObject
381  (
382  mTarget,
383  typeid(mTarget),
384  tTrackingMode,
385  &aIsSaved
386  );
387  iSerializer.writePreElement();
388  iSerializer.saveControl(aSerializeInfo.mObjectId);
389  // 未保存の時のみ保存する
390  if (!aIsSaved)
391  {
392  iSerializer.writePreElement();
393  iSerializer.savePrimitive(mValue);
394  }
395  }
396  }

398  // 回復

400  void load(tBaseSerializer& iSerializer)
401  {
402  mDoSucceed=true;
403  if (tTrackingMode == TrackingMode::etmDefault)
404  {
405  iSerializer.loadPrimitive(mValue);
406  }
407  else
408  {
409  BaseSerializer::AutoRestoreLoad aAutoRestoreLoad(iSerializer);
410  size_t aObjectId;
411  iSerializer.readPreElement();
412  iSerializer.loadControl(aObjectId);
413  bool aIsLoaded;
414  iSerializer.recoverObject
415  (
416  aObjectId,
417  reinterpret_cast<void*&>(mTarget),
418  typeid(mTarget),
419  tTrackingMode,
420  &aIsLoaded
421  );
422  // 未回復の時のみ回復する
423  if (!aIsLoaded)
424  {
425  iSerializer.readPreElement();
426  iSerializer.loadPrimitive(mValue);
427  }
428  }
429 THEOLIZER_INTERNAL_VER_DEBUG((
430  std::cout << "TheolizerNonKeepStep<tPrimitiveType>::load(" << mValue << ") "
431  << (void*)mTarget << std::endl;
